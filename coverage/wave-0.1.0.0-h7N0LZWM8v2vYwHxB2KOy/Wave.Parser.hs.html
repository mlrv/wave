   15 type Parser = P.Parsec Void T.Text

   17 type Ann = P.SourcePos
   18 
   19 runParser :: Parser a -> FilePath -> T.Text -> Either T.Text a
   20 runParser p file src =
   21   let withEof = p <* P.eof
   22       parsed = P.runParser withEof file src
   23    in left (T.pack . P.errorBundlePretty) parsed
   24 
   25 parseLit :: Parser Lit
   26 parseLit =
   27   P.choice
   28     [ LString <$> stringLiteral,
   29       numberLiteral
   30     ]
   31 
   32 sc :: Parser ()
   33 sc =
   34   L.space
   35     P.hspace1
   36     (L.skipLineComment "//")
   37     (L.skipBlockCommentNested "/*" "*/")
   38 
   39 lexeme :: Parser a -> Parser a
   40 lexeme = L.lexeme sc
   41 
   42 symbol :: T.Text -> Parser ()
   43 symbol = void . L.symbol sc
   44 
   45 stringLiteral :: Parser T.Text
   46 stringLiteral =
   47   fmap T.pack $ P.char '\"' *> P.manyTill L.charLiteral (P.char '\"')
   48 
   49 numberLiteral :: Parser Lit
   50 numberLiteral = do
   51   sign <- P.optional $ P.char '-'
   52   int <- P.some P.digitChar
   53   dec <- P.optional $ P.char '.' *> P.some P.digitChar
   54   case dec of
   55     Nothing -> pure $ LInt (read $ maybe [] pure sign <> int)
   56     Just n -> pure $ LFloat (read $ maybe [] pure sign <> int <> "." <> n)
