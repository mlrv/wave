    6 -- We want to translate `pattern -> result` to something like
    7 --
    8 -- if (conditions) {
    9 --   return function (matchersVar*) {
   10 --     return result
   11 --   }(matchersExpr*)

   14 data PatResult = PatResult
   15   { conditions :: [JS.Expr],
   16     matchers :: [(Var, JS.Expr)]
   17   }
   18 
   19 instance Semigroup PatResult where
   20   (<>) (PatResult c1 m1) (PatResult c2 m2) =
   21     PatResult (c1 <> c2) (m1 <> m2)
   22 
   23 instance Monoid PatResult where
   24   mempty = PatResult [] []
