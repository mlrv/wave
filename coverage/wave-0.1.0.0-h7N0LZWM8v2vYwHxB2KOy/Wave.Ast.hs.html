    5 import qualified Data.Text as T

    9 -- the name of a variable
   10 type Var = T.Text
   11 
   12 -- a literal, either an Integer, a Float, or a String
   13 data Lit
   14   = LInt Int
   15   | LFloat Float
   16   | LString T.Text
   17   deriving (Show, Eq)
   18 
   19 -- an expression that evaluates to a value
   20 data Expr a
   21   = EAnnotated a (Expr a)
   22   | ELit Lit
   23   | EVar Var
   24   | EFun [Var] (Sub a)
   25   | EFunCall (Expr a) [Expr a]
   26   | ERecord (Record (Expr a))
   27   | EFfi T.Text [Expr a] -- foreign function interface call
   28   | EVariant (Variant (Expr a)) -- equivalent to data constructors
   29   | ECase (Expr a) [(Pattern, Expr a)]
   30   | ERecordAccess (Expr a) Label
   31   deriving (Show, Eq)
   32 
   33 -- a datatype
   34 data DataType
   35   = DataType Constr [TypeVar] [Variant Type]
   36   deriving (Show, Eq)
   37 
   38 -- a datatype definition or term definition
   39 data Definition a
   40   = TypeDef DataType
   41   | TermDef (TermDef a)
   42   deriving (Show, Eq)
   43 
   44 -- a term definition, either a variable or a function
   45 data TermDef a
   46   = Variable Var (Expr a)
   47   | Function Var [Var] (Sub a)
   48   deriving (Show, Eq)
   49 
   50 -- a statement, either an expression or a term definition
   51 data Statement a
   52   = SExpr (Expr a)
   53   | SDef (TermDef a)
   54   deriving (Show, Eq)
   55 
   56 -- a series of statements
   57 type Sub a = [Statement a]
   58 
   59 -- a pattern
   60 data Pattern
   61   = PWildcard -- match all
   62   | PVar Var -- match all and bind the expression to a variable
   63   | PLit Lit
   64   | PRecord (Record Pattern)
   65   | PVariant (Variant Pattern)
   66   deriving (Show, Eq)
   67 
   68 -- a wave's source file, a collection of definitions
   69 newtype File a
   70   = File [Definition a]
   71   deriving (Show, Eq)
