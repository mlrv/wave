   11 type Builtins = M.Map Var Builtin
   12 
   13 data Builtin = Builtin
   14   { bName :: Var,
   15     bType :: Type,
   16     bImpl :: Impl
   17   }
   18 
   19 data Impl
   20   = Fun T.Text
   21   | BinOp T.Text
   22 
   23 builtinFun :: Var -> Type -> T.Text -> (Var, Builtin)
   24 builtinFun = builtin Fun
   25 
   26 builtinBinOp :: Var -> Type -> T.Text -> (Var, Builtin)
   27 builtinBinOp = builtin BinOp
   28 
   29 builtin :: (T.Text -> Impl) -> Var -> Type -> T.Text -> (Var, Builtin)
   30 builtin c n t i = (n, Builtin n t $ c i)
   31 
   32 builtins :: Builtins
   33 builtins =
   34   M.unions
   35     [ ints,
   36       strings,
   37       bools
   38     ]
   39 
   40 ints :: Builtins
   41 ints =
   42   M.fromList
   43     [ builtinBinOp "add" binOpInt "+",
   44       builtinBinOp "sub" binOpInt "-",
   45       builtinBinOp "mul" binOpInt "*",
   46       builtinBinOp "div" binOpInt "/",
   47       builtinFun "negate" opInt "function (x) { return 0 - x }"
   48     ]
   49   where
   50     binOpInt = TypeFun [tInt, tInt] tInt
   51     opInt = TypeFun [tInt] tInt
   52 
   53 strings :: Builtins
   54 strings =
   55   M.fromList
   56     [builtinBinOp "concat" binOpInt "+"]
   57   where
   58     binOpInt = TypeFun [tString, tString] tString
   59 
   60 bools :: Builtins
   61 bools =
   62   M.fromList
   63     [ builtinBinOp "and" binOpInt "&&",
   64       builtinBinOp "or" binOpInt "||",
   65       builtinFun "not" opInt "function (x) { return !x }"
   66     ]
   67   where
   68     binOpInt = TypeFun [tBool, tBool] tBool
   69     opInt = TypeFun [tBool] tBool
   70 
   71 -- Builtin values
   72 unit :: Expr ()
   73 unit = ERecord M.empty
   74 
   75 true :: Expr ()
   76 true = EVariant $ Variant "True" unit
   77 
   78 false :: Expr ()
   79 false = EVariant $ Variant "False" unit
   80 
   81 -- Builtin types
   82 tUnit :: Type
   83 tUnit = TypeRec []
   84 
   85 tInt :: Type
   86 tInt = TypeCon "Int"
   87 
   88 tFloat :: Type
   89 tFloat = TypeCon "Float"
   90 
   91 tString :: Type
   92 tString = TypeCon "String"
   93 
   94 tBool :: Type
   95 tBool = TypeCon "Bool"
   96 
   97 -- Builtin datatypes
   98 bool :: DataType
   99 bool =
  100   DataType
  101     "Bool"
  102     []
  103     [ Variant "True" tUnit,
  104       Variant "False" tUnit
  105     ]
  106 
  107 maybe :: DataType
  108 maybe =
  109   DataType
  110     "Maybe"
  111     ["a"]
  112     [ Variant "Just" $ TypeVar "a",
