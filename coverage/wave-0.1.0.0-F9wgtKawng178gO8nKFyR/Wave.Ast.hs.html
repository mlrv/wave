    9 -- the name of a variable
   10 type Var = T.Text
   11 
   12 -- a literal, either an Integer, a Float, or a String
   13 data Lit
   14   = LInt Int
   15   | LFloat Float
   16   | LString T.Text
   17   deriving (Show, Eq)
   18 
   19 -- an expression that evaluates to a value
   20 data Expr
   21   = ELit Lit
   22   | EVar Var
   23   | EFun [Var] Sub
   24   | EFunCall Expr [Expr]
   25   | ERecord (Record Expr)
   26   | EFfi T.Text [Expr] -- foreign function interface call
   27   | EVariant (Variant Expr) -- equivalent to data constructors
   28   | ECase Expr [(Pattern, Expr)]
   29   | ERecordAccess Expr Label
   30   deriving (Show, Eq)
   31 
   32 -- a definition, either a variable or a function
   33 data Definition
   34   = Variable Var Expr
   35   | Function Var [Var] Sub
   36   deriving (Show, Eq)
   37 
   38 -- a statement, either an expression or a definition
   39 data Statement
   40   = SExpr Expr
   41   | SDef Definition
   42   deriving (Show, Eq)
   43 
   44 -- a series of statements
   45 type Sub = [Statement]
   46 
   47 -- a pattern
   48 data Pattern
   49   = PWildcard -- match all
   50   | PVar Var -- match all and bind the expression to a variable
   51   | PLit Lit
   52   | PRecord (Record Pattern)
   53   | PVariant (Variant Pattern)
   54   deriving (Show, Eq)
   55 
   56 -- a wave's source file, a collection of definitions
   57 newtype File
   58   = File [Definition]
   59   deriving (Show, Eq)
