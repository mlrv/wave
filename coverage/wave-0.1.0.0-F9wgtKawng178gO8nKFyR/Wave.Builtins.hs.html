    3 module Wave.Builtins where

    5 import qualified Data.Map as M

   10 
   11 type Builtins = M.Map Var Builtin
   12 
   13 data Builtin = Builtin
   14   { bName :: Var,
   15     bType :: Type,
   16     bImpl :: Impl
   17   }
   18 
   19 data Impl
   20   = Fun T.Text
   21   | BinOp T.Text
   22 
   23 builtinFun :: Var -> Type -> T.Text -> (Var, Builtin)
   24 builtinFun = builtin Fun
   25 
   26 builtinBinOp :: Var -> Type -> T.Text -> (Var, Builtin)
   27 builtinBinOp = builtin BinOp
   28 
   29 builtin :: (T.Text -> Impl) -> Var -> Type -> T.Text -> (Var, Builtin)
   30 builtin c n t i = (n, Builtin n t $ c i)
   31 
   32 builtins :: Builtins
   33 builtins =
   34   M.unions
   35     [ ints
   36     ]
   37 
   38 ints :: Builtins
   39 ints =
   40   M.fromList
   41     [ builtinBinOp "add" binOpInt "+",
   42       builtinBinOp "sub" binOpInt "-",
   43       builtinBinOp "mul" binOpInt "*",
   44       builtinBinOp "div" binOpInt "/",
   45       builtinFun "negate" opInt "function (x) { return 0 - x }"
   46     ]
   47   where
   48     binOpInt = TypeFun [tInt, tInt] tInt
   49     opInt = TypeFun [tInt] tInt
   50 
   51 -- Builtin values
   52 unit :: Expr
   53 unit = ERecord M.empty

   55 true :: Expr
