    1 {-# LANGUAGE OverloadedStrings #-}
    2 
    3 module Translate where
    4 
    5 import qualified Data.Map as M
    6 import qualified JS.Ast as JS
    7 import Wave.Ast
    8 
    9 translateFile :: File -> JS.File
   10 translateFile (File defs) = JS.File $ map (JS.SDef . translateDef) defs
   11 
   12 translateDef :: Definition -> JS.Definition
   13 translateDef = \case
   14   Variable var expr -> JS.Variable var (translateExpr expr)
   15   Function var args body -> JS.Function var args (translateSub body)
   16 
   17 translateExpr :: Expr -> JS.Expr
   18 translateExpr = \case
   19   ELit lit -> JS.ELit (translateLit lit)
   20   EVar var -> JS.EVar var
   21   EFun args body -> JS.EFun args (translateSub body)
   22   EFunCall fun args -> JS.EFunCall (translateExpr fun) (map translateExpr args)
   23   ERecord record -> JS.ERecord (fmap translateExpr record)
   24   Effi fun args -> JS.EFunCall (JS.EVar fun) (map translateExpr args)
   25   EVariant kind value ->
   26     JS.ERecord $
   27       M.fromList
   28         [ ("_kind", JS.ELit $ JS.LString kind),
   29           ("_value", translateExpr value)
   30         ]
   31   -- ECase {} -> undefined
   32 
   33 translateLit :: Lit -> JS.Lit
   34 translateLit = \case
   35   LInt int -> JS.LInt int
   36   LFloat float -> JS.LFloat float
   37   LString str -> JS.LString str
   38 
   39 translateSub :: Sub -> JS.Sub
   40 translateSub = map translateStatement
   41 
   42 translateStatement :: Statement -> JS.Statement
   43 translateStatement = \case
   44   SExpr expr -> JS.SExpr (translateExpr expr)
   45   SDef def -> JS.SDef (translateDef def)
