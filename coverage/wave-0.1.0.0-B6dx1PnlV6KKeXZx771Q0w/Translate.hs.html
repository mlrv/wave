   16 -- translate wave's AST to a subset of JS'AST
   17 type Translate a = MonadState Int a
   18 
   19 translate :: (a -> State Int b) -> a -> b
   20 translate tran ast = evalState (tran ast) 0
   21 
   22 genVar :: Translate m => T.Text -> m Var
   23 genVar var = do
   24   n <- get
   25   modify (+ 1)
   26   pure ("_" <> var <> "_" <> T.pack (show n))

   30 translateFile :: Translate m => File -> m JS.File
   31 translateFile (File defs) = do
   32   defs' <- traverse translateDef defs
   33   pure $ JS.File $ map JS.SDef defs' ++ [JS.SExpr $ JS.EFunCall (JS.EVar "main") []]
   34 
   35 translateDef :: Translate m => Definition -> m JS.Definition
   36 translateDef = \case
   37   Variable var expr -> JS.Variable var <$> translateExpr expr
   38   Function var args body -> JS.Function var args <$> translateSub body
   39 
   40 translateExpr :: Translate m => Expr -> m JS.Expr
   41 translateExpr = \case
   42   ELit lit -> pure $ JS.ELit $ translateLit lit
   43   EVar var -> pure $ JS.EVar var
   44   EFun args body -> JS.EFun args <$> translateSub body
   45   EFunCall fun args -> do
   46     fun' <- translateExpr fun
   47     args' <- traverse translateExpr args
   48     pure $ JS.EFunCall fun' args'
   49   ERecord record -> JS.ERecord <$> traverse translateExpr record
   50   EFfi fun args -> JS.EFunCall (JS.EVar fun) <$> traverse translateExpr args
   51   EVariant kind value -> do
   52     value' <- translateExpr value
   53     pure $
   54       JS.ERecord $
   55         M.fromList
   56           [ ("_kind", JS.ELit $ JS.LString kind),
   57             ("_value", value')
   58           ]
   59   ECase expr patterns -> do
   60     expr' <- translateExpr expr
   61     var <- genVar "case"
   62     patterns' <- translatePatterns (JS.EVar var) patterns
   63     pure $
   64       JS.EFunCall
   65         (JS.EFun [var] patterns')
   66         [expr']
   67   ERecordAccess expr label -> do
   68     expr' <- translateExpr expr
   69     pure $
   70       JS.ERecordAccess expr' label
   71 
   72 translatePatterns :: Translate m => JS.Expr -> [(Pattern, Expr)] -> m JS.Sub
   73 translatePatterns var = traverse $ \(pat, expr) -> do
   74   result' <- translateExpr expr
   75   PatResult conds matchers' <- translatePattern var pat
   76   let (matchersV, matchersE) = unzip matchers'
   77   pure $
   78     JS.SIf
   79       (JS.EAnd conds)
   80       [ JS.SRet $
   81           JS.EFunCall
   82             (JS.EFun matchersV [JS.SRet result'])
   83             matchersE
   84       ]
   85 
   86 translatePattern :: Translate m => JS.Expr -> Pattern -> m PatResult
   87 translatePattern expr = \case
   88   PWildcard ->
   89     pure
   90       PatResult
   91         { conditions = [JS.ELit $ JS.LBool True], -- always match
   92           matchers = [] -- but don't bind anything
   93         }
   94   PVar var ->
   95     pure
   96       PatResult
   97         { conditions = [JS.ELit $ JS.LBool True], -- always match
   98           matchers = [(var, expr)] -- and bind the var to the expression
   99         }
  100   PLit lit ->
  101     pure
  102       PatResult
  103         { conditions = [JS.EEqual (JS.ELit $ translateLit lit) expr],
  104           matchers = []
  105         }
  106   PRecord (M.toList -> fields) -> do
  107     fmap mconcat $
  108       for fields $ \(field, pat) ->
  109         translatePattern (JS.ERecordAccess expr field) pat
  110   PVariant tag pat -> do
  111     pat' <- translatePattern (JS.ERecordAccess expr "_value") pat
  112     pure
  113       PatResult
  114         { conditions =
  115             JS.EEqual
  116               (JS.ELit $ JS.LString tag)
  117               (JS.ERecordAccess expr "_kind") :
  118             conditions pat',
  119           matchers = matchers pat'
  120         }
  121 
  122 translateSub :: Translate m => Sub -> m JS.Sub
  123 translateSub = traverse translateStatement
  124 
  125 translateStatement :: Translate m => Statement -> m JS.Statement
  126 translateStatement = \case
  127   SExpr expr -> JS.SExpr <$> translateExpr expr
  128   SDef def -> JS.SDef <$> translateDef def
  129 
  130 translateLit :: Lit -> JS.Lit
  131 translateLit = \case
  132   LInt int -> JS.LInt int
  133   LFloat float -> JS.LFloat float
  134   LString str -> JS.LString str
