   15 -- expl
   16 type TransState = Int
   17 
   18 type Translate a = MonadState TransState a
   19 
   20 translate :: (a -> State TransState b) -> a -> b
   21 translate tran ast = evalState (tran ast) 0
   22 
   23 genVar :: Translate m => T.Text -> m Var
   24 genVar var = do
   25   n <- get
   26   modify (+ 1)
   27   pure ("_" <> var <> "_" <> T.pack (show n))
   28 
   29 translateFile :: Translate m => File -> m JS.File
   30 translateFile (File defs) = do
   31   defs' <- traverse translateDef defs
   32   pure $ JS.File $ map JS.SDef defs'
   33 
   34 translateDef :: Translate m => Definition -> m JS.Definition
   35 translateDef = \case
   36   Variable var expr -> JS.Variable var <$> translateExpr expr
   37   Function var args body -> JS.Function var args <$> translateSub body
   38 
   39 translateExpr :: Translate m => Expr -> m JS.Expr
   40 translateExpr = \case
   41   ELit lit -> pure $ JS.ELit $ translateLit lit
   42   EVar var -> pure $ JS.EVar var
   43   EFun args body -> JS.EFun args <$> translateSub body
   44   EFunCall fun args -> do
   45     fun' <- translateExpr fun
   46     args' <- traverse translateExpr args
   47     pure $ JS.EFunCall fun' args'
   48   ERecord record -> JS.ERecord <$> traverse translateExpr record
   49   EFfi fun args -> JS.EFunCall (JS.EVar fun) <$> traverse translateExpr args
   50   EVariant kind value -> do
   51     value' <- translateExpr value
   52     pure $
   53       JS.ERecord $
   54         M.fromList
   55           [ ("_kind", JS.ELit $ JS.LString kind),
   56             ("_value", value') -- extract pattern stuff in
   57           ]
   58   ECase expr patterns -> do
   59     expr' <- translateExpr expr
   60     var <- genVar "case"
   61     patterns' <- translatePatterns (JS.EVar var) patterns
   62     pure $
   63       JS.EFunCall
   64         (JS.EFun [var] patterns')
   65         [expr']
   66   ERecordAccess expr label -> do
   67     expr' <- translateExpr expr
   68     pure $
   69       JS.ERecordAccess expr' label
   70 
   71 -- translate `pattern -> result` to something like
   72 -- if (conditions) {
   73 --  return function (matchersV) {
   74 --    return result;
   75 --   }(matchersE)
   76 -- }
   77 
   78 data PatResult = PatResult
   79   { conditions :: [JS.Expr],
   80     matchers :: [(Var, JS.Expr)]
   81   }
   82 
   83 instance Semigroup PatResult where
   84   (<>) (PatResult c1 m1) (PatResult c2 m2) =
   85     PatResult (c1 <> c2) (m1 <> m2)
   86 
   87 instance Monoid PatResult where
   88   mempty = PatResult [] []
   89 
   90 translatePatterns :: Translate m => JS.Expr -> [(Pattern, Expr)] -> m JS.Sub
   91 translatePatterns var = traverse $ \(pat, expr) -> do
   92   result' <- translateExpr expr
   93   PatResult conds matchers' <- translatePattern var pat
   94   let (matchersV, matchersE) = unzip matchers'
   95   pure $
   96     JS.SIf
   97       (JS.EAnd conds)
   98       [ JS.SRet $
   99           JS.EFunCall
  100             (JS.EFun matchersV [JS.SRet result'])
  101             matchersE
  102       ]
  103 
  104 translatePattern :: Translate m => JS.Expr -> Pattern -> m PatResult
  105 translatePattern expr = \case
  106   PWildcard ->
  107     pure
  108       PatResult
  109         { conditions = [JS.ELit $ JS.LBool True],
  110           matchers = []
  111         }
  112   PVar var ->
  113     pure
  114       PatResult
  115         { conditions = [JS.ELit $ JS.LBool True],
  116           matchers = [(var, expr)]
  117         }
  118   PLit lit ->
  119     pure
  120       PatResult
  121         { conditions = [JS.EEqual (JS.ELit $ translateLit lit) expr],
  122           matchers = []
  123         }
  124   PRecord (M.toList -> fields) -> do
  125     fmap mconcat $
  126       for fields $ \(field, pat) ->
  127         translatePattern (JS.ERecordAccess expr field) pat
  128   PVariant tag pat -> do
  129     pat' <- translatePattern (JS.ERecordAccess expr "_value") pat
  130     pure
  131       PatResult
  132         { conditions =
  133             JS.EEqual
  134               (JS.ELit $ JS.LString tag)
  135               (JS.ERecordAccess expr "_kind") :
  136             conditions pat',
  137           matchers = matchers pat'
  138         }
  139 
  140 translateLit :: Lit -> JS.Lit
  141 translateLit = \case
  142   LInt int -> JS.LInt int
  143   LFloat float -> JS.LFloat float
  144   LString str -> JS.LString str
  145 
  146 translateSub :: Translate m => Sub -> m JS.Sub
  147 translateSub = traverse translateStatement
  148 
  149 translateStatement :: Translate m => Statement -> m JS.Statement
  150 translateStatement = \case
  151   SExpr expr -> JS.SExpr <$> translateExpr expr
  152   SDef def -> JS.SDef <$> translateDef def
