    1 {-# LANGUAGE OverloadedStrings #-}
    2 
    3 module JS.Pretty where
    4 
    5 import qualified Data.Map as M
    6 import qualified Data.Text as T
    7 import JS.Ast
    8 import Prettyprinter
    9 import Prettyprinter.Render.Text
   10 
   11 pp :: (a -> Doc ann) -> a -> T.Text
   12 pp f = render . f
   13 
   14 render :: Doc a -> T.Text
   15 render = renderStrict . layoutPretty defaultLayoutOptions
   16 
   17 ppFile :: File -> Doc a
   18 ppFile (File file) = vsep $ ppStatement <$> file
   19 
   20 ppRecord :: (a -> Doc ann) -> Record a -> Doc ann
   21 ppRecord p r =
   22   encloseSep "{" "}" ", " $
   23     fmap
   24       (\(k, v) -> surround ": " (pretty $ show k) v)
   25       (M.toList (fmap p r))
   26 
   27 ppLit :: Lit -> Doc a
   28 ppLit = \case
   29   LInt int -> pretty int
   30   LFloat float -> pretty float
   31   LString str -> pretty str
   32   LBool True -> "true"
   33   LBool False -> "false"
   34 
   35 ppSub :: Sub -> Doc a
   36 ppSub sub = vsep (ppStatement <$> sub)
   37 
   38 ppDef :: Definition -> Doc a
   39 ppDef = \case
   40   Variable name expr -> "const" <+> pretty name <+> "=" <+> ppExpr expr <> ";"
   41   Function name args body ->
   42     let arguments = concatWith (surround ", ") (pretty <$> args)
   43      in "const" <+> pretty name <+> "="
   44           <+> vsep
   45             [ "function" <> "(" <> arguments <> ")" <+> "{",
   46               indent 2 "return" <+> ppSub body <> ";",
   47               "}"
   48             ]
   49 
   50 ppStatement :: Statement -> Doc a
   51 ppStatement = \case
   52   SExpr expr -> ppExpr expr
   53   SRet expr -> ppExpr expr
   54   SDef def -> ppDef def
   55 
   56 ppExpr :: Expr -> Doc a
   57 ppExpr = \case
   58   ELit lit -> ppLit lit
   59   EVar var -> pretty var
   60   EFun args body ->
   61     let arguments = concatWith (surround ", ") (pretty <$> args)
   62      in vsep
   63           [ "function" <> "(" <> arguments <> ")" <+> "{",
   64             indent 2 "return" <+> ppSub body <> ";",
   65             "}"
   66           ]
   67   EFunCall fun args ->
   68     let encloseIfNotSimple = if isSimpleExpr fun then id else parens
   69         arguments = concatWith (surround ", ") (ppExpr <$> args)
   70      in encloseIfNotSimple (ppExpr fun) <> "(" <> arguments <> ")"
   71   ERecord record -> ppRecord ppExpr record
   72 
   73 -- need surrounding parens?
   74 isSimpleExpr :: Expr -> Bool
   75 isSimpleExpr = \case
   76   ERecord {} -> False
   77   EFun {} -> False
   78   _ -> True
