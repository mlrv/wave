    1 {-# LANGUAGE OverloadedStrings #-}
    2 
    3 module JS.Pretty where
    4 
    5 import qualified Data.Map as M
    6 import qualified Data.Text as T
    7 import JS.Ast
    8 import Prettyprinter
    9 import Prettyprinter.Render.Text
   10 
   11 pp :: (a -> Doc a) -> a -> T.Text
   12 pp f = render . f
   13 
   14 render :: Doc a -> T.Text
   15 render = renderStrict . layoutPretty defaultLayoutOptions
   16 
   17 ppRecord :: Pretty a => Record a -> Doc a
   18 ppRecord r =
   19   encloseSep "{" "}" ", " $
   20     fmap
   21       (\(k, v) -> surround ": " (pretty $ show k) v)
   22       (M.toList (fmap pretty r))
   23 
   24 ppLit :: Lit -> Doc a
   25 ppLit = \case
   26   LInt int -> pretty int
   27   LFloat float -> pretty float
   28   LString str -> pretty str
   29   LBool True -> "true"
   30   LBool False -> "false"
