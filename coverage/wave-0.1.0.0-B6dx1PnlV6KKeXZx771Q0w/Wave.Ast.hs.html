    8 -- the name of a variable
    9 type Var = T.Text
   10 
   11 -- a literal, either an Integer, a Float, or a String
   12 data Lit
   13   = LInt Int
   14   | LFloat Float
   15   | LString T.Text
   16   deriving (Show, Eq)
   17 
   18 -- a simple text label
   19 type Label = T.Text
   20 
   21 -- a record with keys of type String and values of type `a`, like
   22 -- { "x": 1, "y": 2, "z": 3 }
   23 type Record a =
   24   M.Map Label a
   25 
   26 -- an expression that evaluates to a value
   27 data Expr
   28   = ELit Lit
   29   | EVar Var
   30   | EFun [Var] Sub
   31   | EFunCall Expr [Expr]
   32   | ERecord (Record Expr)
   33   | EFfi T.Text [Expr] -- foreign function interface call
   34   | EVariant T.Text Expr -- equivalent to data constructors
   35   | ECase Expr [(Pattern, Expr)]
   36   | ERecordAccess Expr Label
   37   deriving (Show, Eq)
   38 
   39 -- a definition, either a variable or a function
   40 data Definition
   41   = Variable Var Expr
   42   | Function Var [Var] Sub
   43   deriving (Show, Eq)
   44 
   45 -- a statement, either an expression or a definition
   46 data Statement
   47   = SExpr Expr
   48   | SDef Definition
   49   deriving (Show, Eq)
   50 
   51 -- a series of statements
   52 type Sub = [Statement]
   53 
   54 -- a pattern
   55 data Pattern
   56   = PWildcard -- match all
   57   | PVar Var -- match all and bind the expression to a variable
   58   | PLit Lit
   59   | PRecord (Record Pattern)
   60   | PVariant T.Text Pattern
   61   deriving (Show, Eq)
   62 
   63 -- a wave's source file, a collection of definitions
   64 newtype File
   65   = File [Definition]
   66   deriving (Show, Eq)
